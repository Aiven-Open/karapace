8from karapace.core.instrumentation.tracer import Tracer 

9from threading import Condition 

10 

11 

12class OffsetWatcher: 

13 """Synchronization container for threads to wait until an offset is seen. 

14 

15 This works under the assumption offsets are used only once, which should be 

16 correct as long as no unclean leader election is performed. 

17 """ 

18 

19 def __init__(self) -> None: 

20 # Condition used to protected _greatest_offset, any modifications to that object must 

21 # be performed with this condition acquired 

22 self._condition = Condition() 

23 self._greatest_offset = -1 # Would fail if initially this is 0 as it will be first offset ever. 

24 self._tracer = Tracer() 

25 

26 def greatest_offset(self) -> int: 

27 with self._tracer.get_tracer().start_as_current_span( 

28 self._tracer.get_name_from_caller_with_class(self, self.greatest_offset) 

29 ): 

30 return self._greatest_offset 

31 

32 def offset_seen(self, new_offset: int) -> None: 

33 with self._condition: 

34 self._greatest_offset = max(self._greatest_offset, new_offset) 

35 self._condition.notify_all() 

36 

37 def wait_for_offset(self, expected_offset: int, timeout: float) -> bool: 

38 """Block until expected_offset is seen. 

39 

40 Args: 

41 expected_offset: The message offset generated by the producer. 

42 timeout: How long the caller will wait for the offset in seconds. 

43 """ 

44 with self._condition: 

45 return self._condition.wait_for(lambda: expected_offset <= self._greatest_offset, timeout=timeout)
