65NAME_ANONYMIZABLE_PATTERN = re.compile("[^.]+")

69def anonymize_name(name: str) -> str: 

70 """Anonymize the name. 

71 

72 Name is splitted by dot to logical elements that the whole name consists of. 

73 The element is sha1 hashed. After hashing element is cleaned to conform to Avro name 

74 format of: 

75 * start with [A-Za-z_] 

76 * subsequently contain only [A-Za-z0-9_] 

77 

78 Returns anonymized name. 

79 """ 

80 

81 def anonymize_element(m: re.Match) -> str: 

82 string = m.group() 

83 

84 # Preserve possible invalid characters as a suffix. 

85 invalid_chars = list(set(INVALID_CHARACTER_PATTERN.findall(string))) 

86 invalid_chars.sort() # for consistency 

87 invalid_chars_suffix = "".join(invalid_chars) 

88 

89 digest = hashlib.sha1(string.encode("utf-8")).hexdigest() 

90 

91 # AVRO requires field names to start from a-zA-Z. SHA-1 can start with number. 

92 # Replace the first character with 'a'. 

93 return "a" + digest[1:] + invalid_chars_suffix 

94 

95 return NAME_ANONYMIZABLE_PATTERN.sub(anonymize_element, name)

98Schema: TypeAlias = Union[str, dict[str, Any], list[Any]] 

99 

100 

101def anonymize(input_schema: Schema) -> Schema: 

102 if not input_schema: 

103 return input_schema 

104 elif isinstance(input_schema, str): 

105 if input_schema in ALL_TYPES: 

106 return input_schema 

107 return anonymize_name(input_schema) 

108 elif isinstance(input_schema, list): 

109 return [anonymize(value) for value in input_schema] 

110 elif isinstance(input_schema, dict): 

111 output_schema: dict[str, Any] = {} 

112 for key, value in input_schema.items(): 

113 if key in KEYWORDS: 

114 output_schema[key] = anonymize(value) 

115 elif key == DOC: 

116 pass # Doc attribute may contain non-public and sensitive data 

117 elif key == ORDER: 

118 if value in ORDER_VALID_VALUES: 

119 output_schema[key] = value 

120 else: 

121 output_schema[key] = anonymize(value) 

122 else: 

123 if isinstance(key, str): 

124 key = anonymize_name(key) 

125 output_schema[key] = anonymize(value) 

126 return output_schema 

127 else: 

128 return input_schema # type: ignore[unreachable]
