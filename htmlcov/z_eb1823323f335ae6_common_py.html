28def single_futmap_result(futmap: dict[Any, Future[T]]) -> T: 

29 """Extract the result of a future wrapped in a dict. 

30 

31 Bulk operations of the `confluent_kafka` library's Kafka clients return results 

32 wrapped in a dictionary of futures. Most often we use these bulk operations to 

33 operate on a single resource/entity. This function makes sure the dictionary of 

34 futures contains a single future and returns its result. 

35 """ 

36 (future,) = futmap.values() 

37 return future.result()

40def translate_from_kafkaerror(error: KafkaError) -> Exception: 

41 """Translate a `KafkaError` from `confluent_kafka` to a friendlier exception. 

42 

43 `kafka.errors.for_code` is used to translate the original exception's error code 

44 to a domain specific error class from `aiokafka`. 

45 

46 In some cases `KafkaError`s are created with error codes internal to `confluent_kafka`, 

47 such as various internal error codes for unknown topics or partitions: 

48 `_NOENT`, `_UNKNOWN_PARTITION`, `_UNKNOWN_TOPIC` - these internal errors 

49 have negative error codes that needs to be handled separately. 

50 """ 

51 code = error.code() 

52 if code in ( 

53 KafkaError._NOENT, # pylint: disable=protected-access 

54 KafkaError._UNKNOWN_PARTITION, # pylint: disable=protected-access 

55 KafkaError._UNKNOWN_TOPIC, # pylint: disable=protected-access 

56 ): 

57 return UnknownTopicOrPartitionError() 

58 if code == KafkaError._TIMED_OUT: # pylint: disable=protected-access 58 ↛ 59line 58 didn't jump to line 59 because the condition on line 58 was never true

59 return KafkaTimeoutError() 

60 if code == KafkaError._STATE: # pylint: disable=protected-access 60 ↛ 61line 60 didn't jump to line 61 because the condition on line 60 was never true

61 return IllegalStateError() 

62 if code == KafkaError._RESOLVE: # pylint: disable=protected-access 62 ↛ 63line 62 didn't jump to line 63 because the condition on line 62 was never true

63 return KafkaUnavailableError() 

64 

65 return for_code(code)

68def raise_from_kafkaexception(exc: KafkaException) -> NoReturn: 

69 """Raises a more developer-friendly error from a `KafkaException`. 

70 

71 The `confluent_kafka` library's `KafkaException` is a wrapper around its internal 

72 `KafkaError`. The resulting, raised exception however is coming from 

73 `aiokafka`, due to these exceptions having human-readable names, providing 

74 better context for error handling. 

75 """ 

76 raise translate_from_kafkaerror(exc.args[0]) from exc
